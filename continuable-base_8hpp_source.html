626 class ownership {
627  explicit constexpr ownership(bool acquired, bool frozen)
628  : acquired_(acquired), frozen_(frozen) {}
629 
630 public:
631  constexpr ownership() : acquired_(true), frozen_(false) {}
632  constexpr ownership(ownership const&) = default;
633  ownership(ownership&& right) noexcept
634  : acquired_(right.consume()), frozen_(right.is_frozen()) {}
635  ownership& operator=(ownership const&) = default;
636  ownership& operator=(ownership&& right) noexcept {
637  acquired_ = right.consume();
638  frozen_ = right.is_frozen();
639  return *this;
640  }
641 
642  // Merges both ownerships together
643  ownership operator|(ownership const& right) const noexcept {
644  return ownership(is_acquired() && right.is_acquired(),
645  is_frozen() || right.is_frozen());
646  }
647 
648  constexpr bool is_acquired() const noexcept { return acquired_; }
649  constexpr bool is_frozen() const noexcept { return frozen_; }
650 
651  void release() noexcept {
652  assert(is_acquired() && "Tried to release the ownership twice!");
653  acquired_ = false;
654  }
655  void freeze(bool enabled = true) noexcept {
656  assert(is_acquired() && "Tried to freeze a released object!");
657  frozen_ = enabled;
658  }
659 
660 private:
661  bool consume() noexcept {
662  if (is_acquired()) {
663  release();
664  return true;
665  }
666  return false;
667  }
668 
670  bool acquired_ : 1;
672  bool frozen_ : 1;
673 };

875 template <typename Executor, typename Invoker, typename Callback,
876  typename NextCallback, typename... Args>
877 void packed_dispatch(Executor&& executor, Invoker&& invoker,
878  Callback&& callback, NextCallback&& nextCallback,
879  Args&&... args) {
880 
881  // Create a worker object which when invoked calls the callback with the
882  // the returned arguments.
883  auto work = [
884  invoker = std::forward<Invoker>(invoker),
885  callback = std::forward<Callback>(callback),
886  nextCallback = std::forward<NextCallback>(nextCallback),
887  args = std::make_tuple(std::forward<Args>(args)...)
888  ]() mutable {
889  util::unpack(std::move(args), [&](auto&&... captured_args) {
890  // Just use the packed dispatch method which dispatches the work on
891  // the current thread.
892  packed_dispatch(this_thread_executor_tag{}, std::move(invoker),
893  std::move(callback), std::move(nextCallback),
894  std::forward<decltype(captured_args)>(captured_args)...);
895  });
896  };
897 
898  // Pass the work functional object to the executor
899  std::forward<Executor>(executor)(std::move(work));

1256 template <typename Data>
1257 auto finalize_composition(
1258  continuable_base<Data, strategy_all_tag>&& continuation) {
1259 
1260  auto ownership_ = base::attorney::ownership_of(continuation);
1261 
1262  auto composition = base::attorney::consume_data(std::move(continuation));
1263 
1264  // Merge all signature hints together
1265  auto signature = util::unpack(composition, [](auto&... entries) {
1266  return util::merge(base::hint_of(util::identity_of(entries))...);
1267  });
1268 
1269  return base::attorney::create(
1270  [ signature,
1271  composition = std::move(composition) ](auto&& callback) mutable {
1272  // We mark the current 2-dimensional position through a pair:
1273  // std::pair<size_constant<?>, size_constant<?>>
1274  // ~~~~~~~~~~~~~~~~ ~~~~~~~~~~~~~~~~
1275  // Continuation pos Result pos
1276  auto begin = std::make_pair(util::size_constant_of<0>(),
1277  util::size_constant_of<0>());
1278  auto pack = util::identity_of(composition);
1279  auto end = util::pack_size_of(pack);
1280  auto condition = [=](auto pos) { return pos.first < end; };
1281 
1282  // Create the result submitter which caches all results and invokes
1283  // the final callback upon completion.
1284  auto submitter = make_all_result_submitter(
1285  std::forward<decltype(callback)>(callback), end, signature);
1286 
1287  // Invoke every continuation with it's callback of the submitter
1288  util::static_while(begin, condition, [&](auto current) mutable {
1289  auto entry =
1290  std::move(std::get<decltype(current.first)::value>(composition));
1291 
1292  // This is the length of the arguments of the current continuable
1293  auto arg_size =
1294  util::pack_size_of(base::hint_of(util::identity_of(entry)));
1295 
1296  // The next position in the result tuple
1297  auto next = current.second + arg_size;
1298 
1299  // Invoke the continuation with the associated submission callback
1300  base::attorney::invoke_continuation(
1301  std::move(entry),
1302  submitter->create_callback(current.second, next));
1303 
1304  return std::make_pair(current.first + util::size_constant_of<1>(),
1305  next);
1306  });
1307  },
